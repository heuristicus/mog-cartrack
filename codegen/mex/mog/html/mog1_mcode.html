1function [foreground] = mog(image,count)
 2%MOG Summary of this function goes here
 3%   Detailed explanation goes here
 4% Inputs:
 5%         image:              MxNxC

 7% Outputs

 9persistent models;

11M = size(image,1);
12N = size(image,2);
13C = size(image,3);
14
15K = 3;       %Number of Gaussians
16T = 0.5;      %Background Threshold
17
18
19if isempty(models)

25parfor m=1:M

72        end
