1function [foreground,background] = mog(image,count,alpha)
 2%MOG Summary of this function goes here
 3%   Detailed explanation goes here
 4% Inputs:
 5%         image:              MxNxC
 6%         
 7% Outputs
 8%         foreground:         MxN
 9%         background:         MxNxC
10persistent models;
11
12M = size(image,1);
13N = size(image,2);
14C = size(image,3);
15
16K = 3;       %Number of Gaussians
17T = 0.7;      %Background Threshold
18
19if isempty(models)
20   models = repmat(struct('weight',zeros(K,1), 'mu', zeros(C,K), 'sigma',zeros(K,1)),M,N); 
21end
22
23foreground = zeros(M,N);
24background = zeros(M,N,C);
25
26for m=1:M
27    for n=1:N        
28        X = reshape(image(m,n,:),C,1);
29        %Preallocate
30        prevWeight = zeros(K,1);
31        prevMu = zeros(C,K);
32        prevSigma = zeros(K,1);
33        
34        modelX = models(m,n);        
35        prevWeight = modelX.weight;
36        prevMu = modelX.mu;
37        prevSigma = modelX.sigma;
38        
39        %Initialization (first frame)
40        if count==1
41%         if(isempty(prevWeight) || isempty(prevMu) || isempty(prevSigma))
42              [prevWeight, prevMu, prevSigma] = modelInit(X,K); 
43        end
44        
45        %Update model
46        [weight,mu,sigma] = update(X,prevWeight,prevMu,prevSigma,alpha);
47        
48        models(m,n).weight = zeros(K,1);
49        models(m,n).mu = zeros(C,K);
50        models(m,n).sigma = zeros(K,1);
51        
52        models(m,n).weight = weight;
53        models(m,n).mu = mu;
54        models(m,n).sigma = sigma;
55        
56        %Sort weights/sigma
57        ratio = weight./sigma;
58        [w,index] = sort(ratio,1,'descend');
59        %Calculate B = number of background gaussians
60        sumB = cumsum(w);
61        [~,B] = min(sumB>T);
62        bgIndexes = index(1:B);
63        
64        %Set the background to the weighted average of gaussians
65        bgWeights = weight(bgIndexes); %Bx1
66        bgMu = mu(:,bgIndexes); %CxB
67        background(m,n,:) = (bgMu*bgWeights)./(sum(bgWeights,1));
68       
69        %See if the pixel matches a background gaussian
70        matchingIndex = find(match(X,mu,sigma));
71        if isempty(matchingIndex)
72            matchingIndex = -1;
73        end
74        matching = find(bgIndexes == matchingIndex(1),1);
75        %No match => foreground
76        if isempty(matching)
77            foreground(m,n) = 1;
78        end
79    end    
80end
81background = uint8(background);
82
83end
